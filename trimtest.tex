\def\testdim#1#2{%
    \ifdim#10=#2
        \report{OK: \string#1 = \the\dimexpr#10}%
    \else
        \report{FAIL: \string#1 = \the\dimexpr#10 \space =!= \the\dimexpr#2\relax}%
    \fi
}
